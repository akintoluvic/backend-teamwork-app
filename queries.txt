CREATE TABLE users (
userId SERIAL PRIMARY KEY,
firstName VARCHAR,
lastName VARCHAR,
email VARCHAR NOT NULL,
password VARCHAR NOT NULL,
gender VARCHAR,
jobRole VARCHAR,
department VARCHAR,
address VARCHAR,
userType VARCHAR DEFAULT 'employee'
);

CREATE TABLE articles (
articleId SERIAL PRIMARY KEY,
title VARCHAR NOT NULL,
article VARCHAR NOT NULL,
authorId INTEGER REFERENCES users(userId) ON DELETE CASCADE,
tag VARCHAR NOT NULL,
inapropriate VARCHAR DEFAULT 'no'
);

CREATE TABLE gifs (
gifId SERIAL PRIMARY KEY,
title VARCHAR NOT NULL,
imageUrl VARCHAR NOT NULL,
authorId INTEGER REFERENCES users(userId) ON DELETE CASCADE,
tag VARCHAR NOT NULL,
inapropriate VARCHAR DEFAULT 'no'
);

CREATE TABLE comments (
commentId SERIAL PRIMARY KEY,
comment VARCHAR NOT NULL,
authorId INTEGER REFERENCES users(userId) ON DELETE CASCADE,
inapropriate VARCHAR DEFAULT 'no',
articleId INTEGER REFERENCES articles(articleId) ON DELETE CASCADE,
gifId INTEGER REFERENCES gifs(gifId) ON DELETE CASCADE
);
