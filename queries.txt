
CREATE TABLE users (
userId SERIAL PRIMARY KEY,
firstName VARCHAR(30),
lastName VARCHAR(30),
email VARCHAR(30) NOT NULL,
password VARCHAR(30) NOT NULL,
gender VARCHAR(30),
jobRole VARCHAR(30),
department VARCHAR(30),
address VARCHAR(255),
userType VARCHAR(30) DEFAULT 'employee'
);

CREATE TABLE articles (
articleId SERIAL PRIMARY KEY,
title VARCHAR(30) NOT NULL,
article VARCHAR(255) NOT NULL,
authorId INTEGER REFERENCES users(userId),
tag VARCHAR(30) NOT NULL,
inapropriate VARCHAR(30) DEFAULT 'no'
);

ALTER TABLE articles ALTER COLUMN authorId TYPE INTEGER REFERENCES users(userId) ON DELETE CASCADE;

CREATE TABLE gifs (
gifId SERIAL PRIMARY KEY,
title VARCHAR(30) NOT NULL,
imageUrl VARCHAR(255) NOT NULL,
authorId INTEGER REFERENCES users(userId),
tag VARCHAR(30) NOT NULL,
inapropriate VARCHAR(30) DEFAULT 'no'
);

CREATE TABLE comments (
commentId SERIAL PRIMARY KEY,
comment VARCHAR(255) NOT NULL,
authorId INTEGER REFERENCES users(userId),
inapropriate VARCHAR(30) DEFAULT 'no',
articleId INTEGER REFERENCES articles(articleId),
gifId INTEGER REFERENCES gifs(gifId)
);
